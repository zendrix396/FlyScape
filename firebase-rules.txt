// Firestore rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to read and write their own data
    match /users/{userId} {
      allow create: if request.auth != null;
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow all users to read bookings but only authenticated users to create/update their own
    match /bookings/{bookingId} {
      // Allow anyone to read booking data for pricing and ticket viewing
      allow read: if true;
      
      // Allow authenticated users to create bookings
      allow create: if request.auth != null;
      
      // Allow users to update and delete only their own bookings
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Allow all users to read available flights
    match /flights/{flightId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.admin == true;
    }
  }
}
